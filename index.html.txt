        let currentRole = "all";

        function getRoleBadge(roles) {
            return roles.map(r => {
                if (r === 'am') return '<span class="px-2 py-0.5 bg-blue-100 text-blue-700 rounded text-[9px] font-bold mr-1 uppercase">AM/AAM</span>';
                if (r === 'nvxl') return '<span class="px-2 py-0.5 bg-emerald-100 text-emerald-700 rounded text-[9px] font-bold mr-1 uppercase">NVXL</span>';
                if (r === 'nvpttt') return '<span class="px-2 py-0.5 bg-orange-100 text-orange-700 rounded text-[9px] font-bold mr-1 uppercase">NVPTTT</span>';
                return '';
            }).join('');
        }

        function renderSteps() {
            const container = document.getElementById('process-steps');
            container.innerHTML = '';
            const filtered = currentRole === "all" ? processSteps : processSteps.filter(s => s.appliedTo.includes(currentRole));
            
            if (filtered.length > 0 && !filtered.find(s => s.id === activeStepId)) {
                activeStepId = filtered[0].id;
            }

            filtered.forEach(step => {
                const btn = document.createElement('button');
                btn.className = `step-btn text-left px-4 py-3 rounded-xl border border-slate-200 transition-all font-bold flex flex-col bg-white ${step.id === activeStepId ? 'active' : ''}`;
                btn.innerHTML = `
                    <div class="flex justify-between items-center w-full mb-1 text-start">
                        <span class="text-[11px] uppercase tracking-tighter">${step.id}. ${step.title}</span>
                        <span class="text-lg">${step.icon}</span>
                    </div>
                    <div class="flex flex-wrap justify-start">${getRoleBadge(step.appliedTo)}</div>
                `;
                btn.onclick = () => { activeStepId = step.id; renderSteps(); };
                container.appendChild(btn);
            });
            renderContent();
        }
